Что меняется для 50+ пользователей

8 конкретных изменений в коде и инфраструктуре относительно базового MVP

1

PM2 Cluster Mode — несколько процессов Node.js

Один Node.js-процесс ограничен одним CPU-ядром. При 50 одновременных запросах он создаёт очередь. PM2 запускает по одному процессу на каждое ядро сервера.

ecosystem.config.js
module.exports = { apps: [{ name: 'safetyinspect-api', script: 'src/server.js', instances: 'max', // по числу CPU ядер exec_mode: 'cluster', // shared port, round-robin max_memory_restart: '512M', env: { NODE_ENV: 'production', PORT: 3000 } }] };
2

Пул соединений PostgreSQL (pg-pool)

Без пула каждый запрос открывает новое соединение к БД. При 50 пользователях — 50 одновременных соединений. Пул управляет ними переиспользуя существующие.

src/db/pool.js
const { Pool } = require('pg'); const pool = new Pool({ host: process.env.DB_HOST, database: process.env.DB_NAME, user: process.env.DB_USER, password: process.env.DB_PASS, max: 20, // макс. соединений на процесс idleTimeoutMillis: 30000, connectionTimeoutMillis: 2000, }); module.exports = pool;
3

Redis для JWT-сессий

При нескольких Node.js-процессах сессии нельзя хранить в памяти одного процесса. Redis — общее хранилище для всех воркеров. Также кешируем токены чтобы не проверять подпись каждый раз.

src/middleware/auth.js
const redis = require('../redis'); async function verifyToken(req, res, next) { const token = req.headers.authorization?.split(' ')[1]; // Проверяем кеш (избегаем crypto-операции) const cached = await redis.get(`jwt:${token}`); if (cached) { req.user = JSON.parse(cached); return next(); } const payload = jwt.verify(token, process.env.JWT_SECRET); await redis.setex(`jwt:${token}`, 3600, JSON.stringify(payload)); req.user = payload; next(); }
4

Nginx — балансировщик и Rate Limiting

Nginx принимает все запросы и распределяет по Node.js процессам. Rate limiting защищает от случайной перегрузки (например, баг в клиенте отправляет 1000 синхронизаций).

nginx/nginx.conf
upstream api_cluster { least_conn; # к наименее нагруженному server 127.0.0.1:3001; server 127.0.0.1:3002; server 127.0.0.1:3003; keepalive 32; } limit_req_zone $binary_remote_addr zone=api:10m rate=30r/s; server { listen 443 ssl http2; location /api/ { limit_req zone=api burst=50 nodelay; proxy_pass http://api_cluster; proxy_http_version 1.1; proxy_set_header Connection ""; } }
5

Bull Queue для загрузки фото

50 инспекторов могут одновременно закончить инспекции и начать загружать фото. Очередь Bull (через Redis) обрабатывает их по 5-10 штук параллельно, не перегружая диск и сеть.

src/queues/photoQueue.js
const Queue = require('bull'); const photoQueue = new Queue('photo-upload', { redis: redisConfig }); // Producer: при получении фото от клиента photoQueue.add({ inspectionId, photoBase64, violationId }, { attempts: 3, backoff: 5000 } ); // Consumer: обрабатывает 5 фото параллельно photoQueue.process(5, async (job) => { const { photoBase64, violationId } = job.data; const url = await minioClient.upload(photoBase64); await db.query( 'UPDATE violations SET photo_url=$1 WHERE id=$2', [url, violationId] ); });
6

WebSocket через Redis Pub/Sub

Дашборд руководителя обновляется в реальном времени. Когда инспектор на процессе #1 завершает инспекцию — руководитель подключён к процессу #3. Redis Pub/Sub передаёт событие между воркерами.

src/realtime/ws.js
const { createAdapter } = require('@socket.io/redis-adapter'); const pubClient = redis.duplicate(); const subClient = redis.duplicate(); io.adapter(createAdapter(pubClient, subClient)); // Когда инспекция завершена — публикуем событие io.to(`branch:${branchId}`).emit('inspection:completed', { inspectorName, violationsCount, section }); // Клиент (Flutter Web / браузер) socket.on('inspection:completed', (data) => { updateDashboard(data); // живое обновление без перезагрузки });
7

MinIO вместо локального диска для фото

При нескольких Node.js процессах файл сохранённый процессом #1 недоступен через процесс #3. MinIO — S3-совместимое on-premise хранилище, доступное всем воркерам.

src/storage/minio.js
const Minio = require('minio'); const minio = new Minio.Client({ endPoint: 'minio.local', port: 9000, useSSL: false, accessKey: process.env.MINIO_ACCESS, secretKey: process.env.MINIO_SECRET, }); async function uploadPhoto(buffer, filename) { await minio.putObject('violations', filename, buffer); return `http://minio.local/violations/${filename}`; }
8

Индексы PostgreSQL для быстрых запросов дашборда

При 50 пользователях в базе быстро накапливаются тысячи записей. Без индексов запрос дашборда будет делать full-table scan на каждое открытие.

migrations/add_indexes.sql
-- Дашборд: инспекции за период CREATE INDEX idx_inspections_branch_date ON inspections(branch_id, conducted_at DESC); -- Нарушения по статусу CREATE INDEX idx_violations_status ON violations(status, created_at DESC); -- Мероприятия: просроченные CREATE INDEX idx_actions_due_status ON corrective_actions(due_date, status) WHERE status NOT IN ('closed');

⚡

Производительность

50 одновременных запросов обрабатываются за <200 мс. PDF генерируется ≤5 с даже при пиковой нагрузке.

🛡️

Надёжность

Падение одного Node.js-воркера не роняет систему. Nginx автоматически исключает упавший процесс из пула.

📡

Офлайн не ломается

50 человек могут синхронизировать офлайн-данные одновременно. Bull Queue ставит их в очередь, ни одна инспекция не теряется.
